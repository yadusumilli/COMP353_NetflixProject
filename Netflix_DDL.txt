create table entertainment (
        ShowID            int,
        Title             varchar(500), 
        DateAdded         varchar(20), 
        ReleaseYear       int,          
        Rating            char,
        Description       varchar(500), 
        EntertainmentType char, 
    primary key (ShowID)
) ;

create table entertainmentgenre  (
        ShowID         int,
        GenreType      char,
    primary key (ShowID, GenreType),
    foreign key (showID) references entertainment (ShowId)
) ;

create table entertainmentcast  (
        ShowID         int,
        CastName       char,
    primary key (ShowID, CastName),
    foreign key (showID) references entertainment (ShowId)
) ;

create table country  (
        ShowID         int,
        CountryName    char,
    primary key (ShowID, CountryName),
    foreign key (showID) references entertainment (ShowId)
) ;

create table directedby (
        ShowID         int,
        DirectorName       char,
    primary key (ShowID, DirectorName)
) ;


create table director (
        DirectorName       char,
    primary key (DirectorName)
) ;

create table entertainmenttype (
        Movie       char,
        TVShow      char
) ;
